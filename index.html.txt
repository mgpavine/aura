<script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-database.js"></script>

<style>
body {
    background: black;
    color: #00ffff;
    font-family: monospace;
    text-align: center;
}

.turma {
    margin: 10px;
    padding: 15px;
    border-radius: 12px;
    position: relative;
    overflow: hidden;
    color: white;
}

.explosao {
    position: absolute;
    width: 10px;
    height: 10px;
    background: yellow;
    border-radius: 50%;
    animation: boom 0.6s ease-out forwards;
}

@keyframes boom {
    0% { transform: scale(1); opacity: 1; }
    100% { transform: scale(15); opacity: 0; }
}

button {
    margin: 5px;
    padding: 8px;
    border-radius: 6px;
    border: none;
}

<body>

<h1>⚡ AURA DAS TURMAS ⚡</h1>

<input type="password" id="senha" placeholder="Senha">
<button onclick="login()">Entrar</button>

<div id="turmas"></div>

<h2>🏆 Ranking</h2>
<ol id="ranking"></ol>

<script>

let admin = false;
const SENHA = "1234";

// SONS
const somGanho = new Audio("https://assets.mixkit.co/sfx/preview/mixkit-achievement-bell-600.mp3");
const somPerda = new Audio("https://assets.mixkit.co/sfx/preview/mixkit-player-losing-or-failing-2042.mp3");

// 🔥 COLE SUA CONFIG DO FIREBASE AQUI
const firebaseConfig = {
  apiKey: "mgp377377377",
  authDomain: "aurometro-ceae0.firebaseapp.com",
  databaseURL: "https://aurometro-ceae0-default-rtdb.firebaseio.com",
  projectId: "aurometro-ceae0",
};

firebase.initializeApp(firebaseConfig);
const db = firebase.database();
const ref = db.ref("turmas");

const cores = [
"#ff006e","#8338ec","#3a86ff","#06d6a0",
"#ffbe0b","#fb5607","#ff006e","#00f5d4",
"#9b5de5","#f15bb5","#00bbf9","#00f5d4"
];

const listaTurmas = [
"1º A","1º B","1º C","1º SJ",
"2º A","2º B","2º C","2º SJ",
"3º A","3º B","3º C","3º SJ"
];

ref.once("value", snap => {
    if (!snap.exists()) {
        let dados = listaTurmas.map((t,i) => ({
            nome: t,
            aura: 0,
            cor: cores[i],
            historico: []
        }));
        ref.set(dados);
    }
});

function login() {
    if (document.getElementById("senha").value === SENHA) {
        admin = true;
        alert("Modo professor ⚡");
    }
}

function alterar(i, v) {
    if (!admin) return alert("Somente professor");

    ref.once("value", snap => {
        let turmas = snap.val();

        turmas[i].aura += v;
        turmas[i].historico.push({
            valor: v,
            data: new Date().toLocaleTimeString()
        });

        ref.set(turmas);

        explosao(i);

        if (v > 0) somGanho.play();
        else somPerda.play();
    });
}

function explosao(index) {
    let divs = document.getElementsByClassName("turma");
    let boom = document.createElement("div");
    boom.className = "explosao";
    boom.style.left = "50%";
    boom.style.top = "50%";
    divs[index].appendChild(boom);
    setTimeout(() => boom.remove(), 600);
}

ref.on("value", snap => {
    let turmas = snap.val();

    let div = document.getElementById("turmas");
    let ranking = document.getElementById("ranking");

    div.innerHTML = "";
    ranking.innerHTML = "";

    turmas.forEach((t,i) => {
        div.innerHTML += `
        <div class="turma" style="background:${t.cor}">
            <h3>${t.nome}</h3>
            <p>${t.aura}</p>

            ${admin ? `
                <button onclick="alterar(${i},10)">+10</button>
                <button onclick="alterar(${i},-10)">-10</button>
            ` : ""}
        </div>
        `;
    });

    let ordenado = [...turmas].sort((a,b)=>b.aura-a.aura);

    ordenado.forEach(t=>{
        ranking.innerHTML += `<li>${t.nome} - ${t.aura}</li>`;
    });
});

if ('serviceWorker' in navigator) {
    navigator.serviceWorker.register('service-worker.js');
}

</script>

</body>
</html>
